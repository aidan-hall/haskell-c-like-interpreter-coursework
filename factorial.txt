main() {
  x = 0;
  while (x <= 5) {
    factorial(x);
    factorial_iterative(x);
    x = x + 1;
  }
}

factorial(n) {
  if (n < 0)
    return 0;
  else if (n <= 1)
    return 1;
  else
    return n * factorial(n - 1);
}

factorial_iterative(n) {
  // Necessary to handle special case.
  if (n < 0)
    return 0;

  prod = 1;
  while (n > 0) {
    prod = prod * n;
    n = n - 1;
  }
  return prod;
}
